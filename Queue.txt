#include <stdio.h>
#include <stdlib.h> // for exit()

#define MAX 50

int queue_array[MAX];
int front = -1;
int rear = -1;

int IsEmpty() {
    if (front == -1 || front > rear)
        return 1;  // true
    else
        return 0;  // false
}

int IsFull() {
    if (rear == MAX - 1)
        return 1;
    else
        return 0;
}

void enqueue(int x) {
    if (IsFull()) {
        printf("Queue is Full\n");
        return;
    } else {
        if (IsEmpty()) {
            front = 0;
            rear = 0;
        } else {
            rear = rear + 1;
        }
        queue_array[rear] = x;
        printf("Inserted %d\n", x);
    }
}

void dequeue() {
    if (IsEmpty()) {
        printf("Queue is Empty\n");
        return;
    } else {
        printf("Deleted Element is: %d\n", queue_array[front]);
        front = front + 1;
        if (front > rear) {  // Reset queue when empty
            front = -1;
            rear = -1;
        }
    }
}

void display() {
    if (IsEmpty()) {
        printf("Queue is empty\n");
    } else {
        printf("Queue is: ");
        for (int i = front; i <= rear; i++)
            printf("%d ", queue_array[i]);
        printf("\n");
    }
}

int main() {
    int choice, value;

    while (1) {
        printf("\n1.Insert\n");
        printf("2.Delete\n");
        printf("3.Display\n");
        printf("4.Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
        case 1:
            printf("Enter the element to insert: ");
            scanf("%d", &value);
            enqueue(value);
            break;
        case 2:
            dequeue();
            break;
        case 3:
            display();
            break;
        case 4:
            exit(0);
        default:
            printf("Invalid choice\n");
        }
    }

    return 0;
}
